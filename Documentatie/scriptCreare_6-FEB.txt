-- 1. ABONAMENTE
CREATE TABLE ABONAMENTE (
    AbonamentID  NUMBER(10) PRIMARY KEY,
    TipAbonament VARCHAR2(100) NOT NULL,
    Pret         NUMBER(10, 2) NOT NULL CHECK (Pret > 0)
);

-- 2. ANTRENORI
CREATE TABLE ANTRENORI (
    AntrenorID   NUMBER(10) PRIMARY KEY,
    Nume         VARCHAR2(50) NOT NULL,
    Prenume      VARCHAR2(50) NOT NULL,
    Specializare VARCHAR2(100)
);

-- 3. MEMBRI
CREATE TABLE MEMBRI (
    MembruID    NUMBER(10) PRIMARY KEY,
    Nume        VARCHAR2(50) NOT NULL,
    Prenume     VARCHAR2(50) NOT NULL,
    Email       VARCHAR2(100) UNIQUE,
    AbonamentID NUMBER(10) NOT NULL,
    CONSTRAINT FK_MEMBRI_ABONAMENT FOREIGN KEY (AbonamentID) REFERENCES ABONAMENTE(AbonamentID) ON DELETE CASCADE
);

-- 4. MENTORAT (M:N intre Antrenori si Membri)
CREATE TABLE MENTORAT (
    AntrenorID NUMBER(10) NOT NULL,
    MembruID   NUMBER(10) NOT NULL,
    CONSTRAINT PK_MENTORAT PRIMARY KEY (AntrenorID, MembruID),
    CONSTRAINT FK_MENTORAT_ANTRENOR FOREIGN KEY (AntrenorID) REFERENCES ANTRENORI(AntrenorID) ON DELETE CASCADE,
    CONSTRAINT FK_MENTORAT_MEMBRU FOREIGN KEY (MembruID) REFERENCES MEMBRI(MembruID) ON DELETE CASCADE
);

-- 5. TIPURI ANTRENAMENT (Sursa unica pentru NumeWOD si Descriere)
CREATE TABLE TIPURI_ANTRENAMENT (
    TipID       NUMBER(10) PRIMARY KEY,
    NumeWOD     VARCHAR2(100) NOT NULL,
    Descriere   VARCHAR2(1000)
);

-- 6. CLASE (FARA coloane redundante!)
CREATE TABLE CLASE (
    ClasaID    NUMBER(10) PRIMARY KEY,
    AntrenorID NUMBER(10) NOT NULL,
    TipID      NUMBER(10) NOT NULL,
    DataOra    DATE NOT NULL,
    CONSTRAINT FK_CLASE_ANTRENOR FOREIGN KEY (AntrenorID) REFERENCES ANTRENORI(AntrenorID) ON DELETE CASCADE,
    CONSTRAINT FK_CLASE_TIP FOREIGN KEY (TipID) REFERENCES TIPURI_ANTRENAMENT(TipID) ON DELETE CASCADE
);

-- 7. INSCRIERI (M:N intre Membri si Clase)
CREATE TABLE INSCRIERI (
    MembruID NUMBER(10) NOT NULL,
    ClasaID  NUMBER(10) NOT NULL,
    CONSTRAINT PK_INSCRIERI PRIMARY KEY (MembruID, ClasaID),
    CONSTRAINT FK_INSCRIERI_MEMBRU FOREIGN KEY (MembruID) REFERENCES MEMBRI(MembruID) ON DELETE CASCADE,
    CONSTRAINT FK_INSCRIERI_CLASA FOREIGN KEY (ClasaID) REFERENCES CLASE(ClasaID) ON DELETE CASCADE
);

-- 8. COMPETITII
CREATE TABLE COMPETITII (
    CompetitieID NUMBER(10) PRIMARY KEY,
    Nume         VARCHAR2(100) NOT NULL,
    Data         DATE NOT NULL,
    Taxa         NUMBER(10, 2) DEFAULT 0
);

-- 9. PARTICIPARI_COMPETITIE (M:N)
CREATE TABLE PARTICIPARI_COMPETITIE (
    CompetitieID NUMBER(10) NOT NULL,
    MembruID     NUMBER(10) NOT NULL,
    LoculObtinut NUMBER(3) DEFAULT 0,
    CONSTRAINT PK_PARTICIPARI_COMP PRIMARY KEY (CompetitieID, MembruID),
    CONSTRAINT FK_PART_COMP FOREIGN KEY (CompetitieID) REFERENCES COMPETITII(CompetitieID) ON DELETE CASCADE,
    CONSTRAINT FK_PART_MEMBRU FOREIGN KEY (MembruID) REFERENCES MEMBRI(MembruID) ON DELETE CASCADE
);

-- 10. ECHIPAMENTE
CREATE TABLE ECHIPAMENTE (
    EchipamentID     NUMBER(10) PRIMARY KEY,
    NumeEchipament   VARCHAR2(100) NOT NULL,
    CantitateTotala  NUMBER(5) CHECK (CantitateTotala >= 0)
);

-- 11. NECESAR_ECHIPAMENT (M:N intre TipAntrenament si Echipamente)
CREATE TABLE NECESAR_ECHIPAMENT (
    TipID        NUMBER(10) NOT NULL,
    EchipamentID NUMBER(10) NOT NULL,
    CONSTRAINT PK_NECESAR_ECHIPAMENT PRIMARY KEY (TipID, EchipamentID),
    CONSTRAINT FK_NECESAR_TIP FOREIGN KEY (TipID) REFERENCES TIPURI_ANTRENAMENT(TipID) ON DELETE CASCADE,
    CONSTRAINT FK_NECESAR_ECHIPAMENT FOREIGN KEY (EchipamentID) REFERENCES ECHIPAMENTE(EchipamentID) ON DELETE CASCADE
);

-- 12. PRODUSE
CREATE TABLE PRODUSE (
    ProdusID   NUMBER(10) PRIMARY KEY,
    NumeProdus VARCHAR2(100) NOT NULL,
    PretCurent NUMBER(10, 2) NOT NULL,
    Stoc       NUMBER(5) DEFAULT 0
);

-- 13. ACHIZITII (Tranzactii produse)
CREATE TABLE ACHIZITII (
    AchizitieID NUMBER(10) PRIMARY KEY,
    MembruID    NUMBER(10) NOT NULL,
    ProdusID    NUMBER(10) NOT NULL,
    Data        DATE DEFAULT SYSDATE,
    Cantitate   NUMBER(5) CHECK (Cantitate > 0),
    CONSTRAINT FK_ACHIZITII_MEMBRU FOREIGN KEY (MembruID) REFERENCES MEMBRI(MembruID) ON DELETE CASCADE,
    CONSTRAINT FK_ACHIZITII_PRODUS FOREIGN KEY (ProdusID) REFERENCES PRODUSE(ProdusID) ON DELETE CASCADE
);
